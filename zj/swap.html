function swapFn1(a, b) {
    console.log(a, b);
    a += b;
    b = a - b;
    a -= b;
    console.log(a, b);
  }

  function swapFn2(a, b) {
    console.log(a, b);
    a ^= b;
    b ^= a;
    a ^= b;
    console.log(a, b);
  }

  function swapFn3(a, b) {
    console.log(a, b);
    [a, b] = [b, a];
    console.log(a, b);
  }

  let x1 = 1;
  let x2 = 2;
  let x3 = 3;
  let x4 = 4;
  let x5 = 5;
  let x6 = 6;

  console.log(x1, x2, x3, x4, x5, x6)
  swapFn1(x1, x2)
  swapFn2(x3, x4)
  swapFn3(x5, x6)
  console.log(x1, x2, x3, x4, x5, x6)
