function trim(str) {
    let start, end
    for (let i = 0; i < str.length; i++) {
      if (str[i] !== ' ') {
        start = i
        break
      }
    }
    for (let i = str.length - 1; i > 0; i--) {
      if (str[i] !== ' ') {
        end = i
        break
      }
    }

    return str.substring(start, end + 1)
  }


  function trimExp(str) {
    return str.replace(/(^\s*)|(\s*$)/g, '')
  }
